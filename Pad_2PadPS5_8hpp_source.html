133 void update() noexcept
134 {
135 Traits::MaybeInvokeUpdate<ReaderType>(*this);
136
137 if (const auto message = ReaderType::getMessage())
138 {
139 isConnected = true;
140
141 triangle.update(message->triangle);
142 circle.update(message->circle);
143 cross.update(message->cross);
144 square.update(message->square);
145
146 up.update(message->up);
147 right.update(message->right);
148 down.update(message->down);
149 left.update(message->left);
150
151 l1.update(message->l1);
152 r1.update(message->r1);
153 l2.update(message->l2);
154 r2.update(message->r2);
155 l3.update(message->l3);
156 r3.update(message->r3);
157
158 create.update(message->create);
159 option.update(message->option);
160 touch.update(message->touch);
161 mic.update(message->mic);
162 ps.update(message->ps);
163
164 // -128 ~ 127(int8_t) -> -255 ~ 255(int16_t)
165 const auto decodeStick = [](int8_t raw) -> int16_t
166 {
167 return raw * 2 + 1; // (raw + 128) * 2 - 255
168 };
169
170 leftStick = {
171 CutDeadZone(decodeStick(message->analogLeftX), 20),
172 CutDeadZone(decodeStick(message->analogLeftY), 20),
173 };
174 rightStick = {
175 CutDeadZone(decodeStick(message->analogRightX), 20),
176 CutDeadZone(decodeStick(message->analogRightY), 20),
177 };
178 }
179 else
180 {
181 isConnected = false;
182
183 triangle.update(false);
184 circle.update(false);
185 cross.update(false);
186 square.update(false);
187
188 up.update(false);
189 right.update(false);
190 down.update(false);
191 left.update(false);
192
193 l1.update(false);
194 r1.update(false);
195 l2.update(false);
196 r2.update(false);
197 l3.update(false);
198 r3.update(false);
199
200 create.update(false);
201 option.update(false);
202 touch.update(false);
203 mic.update(false);
204
205 leftStick.clear();
206 rightStick.clear();
207 }
208 }

275 static double CutDeadZone(double value, uint8_t deadZone)
276 {
277 if (value > deadZone)
278 return 255 * (value - deadZone) / (255 - deadZone);
279 else if (value < -deadZone)
280 return 255 * (value + deadZone) / (255 - deadZone);
281 else
282 return 0.0;
283 };
