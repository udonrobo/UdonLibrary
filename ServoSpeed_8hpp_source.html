64  last_angle = next_angle; // 前周期のデータを保管しておく
65  move_angle =

67  if (move_angle == 0)

74  value = Udon::Constrain(speed / static_cast<float>(loopCycle),
75  Udon::Abs(move_angle) * -1, Udon::Abs(move_angle)) *

77  next_angle += value;
