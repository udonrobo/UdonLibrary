33  struct IsPrintableImpl
34  {
35  using ResultType = bool;
36 
37  template <typename... Args>
38  constexpr ResultType operator()(Args&&... args) const noexcept
39  {
40  return argsUnpack(std::forward<Args>(args)...);
41  }

43  private:
44  // 可変長引数展開
45  template <typename Head, typename... Tail>
46  constexpr ResultType argsUnpack(Head&& head, Tail&&... tail) const noexcept
47  {
48  // 全メンバが出力可能か判定し、論理積を得る
49  return Test<RemoveReferenceT<Head>>::test(*this, std::forward<Head>(head)) and argsUnpack(std::forward<Tail>(tail)...);
50  }
51 
52  // 可変長引数展開 (終端)
53  constexpr ResultType argsUnpack() const noexcept { return true; }
54

56  // T が出力可能な型か判定するテスト
57  // 部分特殊化を用いる
58  template <typename T, typename = void>
59  struct Test
60  {
61  static constexpr bool test(...)
62  {
63  //static_assert(AlwaysFalse<T>::value, "Udon::Print: T is not printable." __FUNCTION__);
64  return false;
65  }
66  };

68  // 列挙型は基底型が出力可能であるとき出力可能
69  template <typename Enum>
70  struct Test<Enum, EnableIfVoidT<IsEnum<Enum>::value>>
71  {
72  static constexpr bool test(...) { return true; }
73  };
74 
75  // 配列型(文字配列除く)は要素が出力可能であるとき出力可能
76  template <typename Array>
77  struct Test<Array, EnableIfVoidT<
78  IsArray<Array>::value // 配列である
79  and not IsCString<Array>::value // C言語スタイル文字配列でない
80  and not IsOutputStreamable<OutputStream, Array>::value // ストリームへの出力が不可能
81  >>
82  {
83  template <typename T>
84  static constexpr bool test(const IsPrintableImpl& self, T&& array)
85  {
86  return Test<typename std::remove_extent<Array>::type>::test(self, *array); // 要素が出力可能である場合
87  }
88  };

103  >>

106  };

108  // enumerate が存在すれば出力可能
109  template <typename Enumeratable>
110  struct Test<Enumeratable, EnableIfVoidT<
111  HasMemberFunctionEnumerate<Enumeratable>::value // enumerate 関数が存在する
112  and not HasMemberFunctionShow<Enumeratable>::value // show 関数が存在しない
113  and not IsOutputStreamable<OutputStream, Enumeratable>::value // ストリームへの出力が不可能
114  >>
115  {
116  template <typename T>

118  {
119  return e.enumerate(tester); // enumerate 関数が true を返した場合
120  }

128  template <typename OutputStream>
129  class Printer
130  {
131  public:
132  Printer(OutputStream& stream, bool delimiterEnable)
133  : stream(stream)
134  , delimiterEnable(delimiterEnable)
135  {
136  }
137 
138  using ResultType = void;
139 
140  template <typename... Args>
141  constexpr ResultType operator()(Args&&... args) const
142  {
143  const_cast<Printer&>(*this).argsUnpack(std::forward<Args>(args)...);
144  }
145 
146  private:
147  OutputStream& stream; // 出力先
148 
149  bool delimiterEnable; // 区切り文字を出力するか
150 
152  template <typename Head, typename... Tail>
153  ResultType argsUnpack(Head&& head, Tail&&... tail)
154  {
155  print(std::forward<Head>(head));
156  if (delimiterEnable and sizeof...(Tail) > 0)
157  stream << ", ";
158  argsUnpack(std::forward<Tail>(tail)...);
159  }
160 
163  template <typename Head>
164  ResultType argsUnpack(Head&& head)
165  {
166  print(std::forward<Head>(head));
167  }
168 
170  // ResultType argsUnpack() {}
171 
172  private:
174  template <typename Enum, EnableIfNullptrT<IsEnum<RemoveReferenceT<Enum>>::value> = nullptr>
175  ResultType print(Enum&& e)
176  {
177  print(static_cast<typename std::underlying_type<RemoveReferenceT<Enum>>::type>(e));
178  }
179 
182  template <typename Array, EnableIfNullptrT<
183  IsArray<RemoveReferenceT<Array>>::value // 配列である
184  and not IsCString<RemoveReferenceT<Array>>::value // C言語スタイル文字列は配列として扱わない
185  and not IsOutputStreamable<OutputStream, RemoveReferenceT<Array>>::value // ストリームへの出力が不可能
186  > = nullptr>
187  ResultType print(Array&& array)
188  {
189  stream << "{ ";
190  bool isFirst = true;
191  for (auto&& e : array)
192  {
193  if (isFirst)
194  isFirst = false;
195  else
196  stream << ", ";
197  print(e);
198  }
199  stream << " }";
200  }
201 
202  // 出力用関数の優先度
203  // 1. show()
204  // 2. operator<<()
205  // 3. enumerate()
206 
208  template <typename Printable, EnableIfNullptrT<HasMemberFunctionShow<RemoveReferenceT<Printable>>::value> = nullptr>
209  ResultType print(Printable&& printable)
210  {
211  printable.show();
212  }
213 
215  template <typename OutputStreamable, EnableIfNullptrT<
216  IsOutputStreamable<OutputStream, RemoveReferenceT<OutputStreamable>>::value // ストリームへ出力可能
217  and not HasMemberFunctionShow<RemoveReferenceT<OutputStreamable>>::value // show が存在しない
218  > = nullptr>
219  ResultType print(OutputStreamable&& outputStreamable)
220  {
221  stream << std::forward<OutputStreamable>(outputStreamable);
222  }
223 
225  template <typename Enumeratable, EnableIfNullptrT<
226  HasMemberFunctionEnumerate<RemoveReferenceT<Enumeratable>>::value // enumerate が存在
227  and not HasMemberFunctionShow<RemoveReferenceT<Enumeratable>>::value // show が存在しない
228  and not IsOutputStreamable<OutputStream, RemoveReferenceT<Enumeratable>>::value // ストリームへ出力可能でない
229  > = nullptr>
230  ResultType print(Enumeratable&& enumerable)
231  {
232  stream << "{ ";
233  enumerable.enumerate(*this);
234  stream << " }";
235  }
