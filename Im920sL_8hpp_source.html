161 bool sendUpdate()
162 {
163 if (not nodeNum)
164 {
165 return false;
166 }
167
168 const int sendTimeMs = 52; // 連続で送信する場合、52ms送信休止時間が必要 (説明書 ７－３（２）送信休止時間 参照)
169
170 if (millis() - txNode->transmitMs < sendTimeMs)
171 {
172 return false;
173 }
174
175 // データ送信
176 {
177 Udon::Printf(uart, "TXDU %04d ", *nodeNum);
178
179 Udon::BitPack(txNode->data, txNode->data + txNode->size, [this](uint8_t data)
180 { uart.write(data); });
181
182 uart.print("\r\n");
183 }
184
185 if (uart.readStringUntil('\n') == "OK\r")
186 {
187 txNode->transmitMs = millis();
188 }
189
190 return true;
191 }

193 bool receiveUpdate()
194 {
195 // header = [Dummy: 2byte] + [,] + [Node number: 4byte] + [,] + [RSSI: 2byte] + [:]
196 constexpr int HeaderSize = 2 + 1 + 4 + 1 + 2 + 1;
197
198 // footer = [\r] + [\n]
199 constexpr int FooterSize = 1 + 1;
200
201 // FrameSize = header + data + footer
202 const int FrameSize = HeaderSize + Udon::BitPackedSize(rxNode->size) + FooterSize;
203
204 // 受信バッファにデータがない場合は何もしない
205 if (uart.available() < FrameSize)
206 {
207 return false;
208 }
209
210 // ヘッダー読み捨て
211 for (int i = 0; i < HeaderSize; ++i)
212 {
213 uart.read();
214 }
215
216 // データ読み込み
217 if (not Udon::BitUnpack(rxNode->data, rxNode->data + rxNode->size, [this]() -> uint8_t
218 { return uart.read(); }))
219 {
220 // 先頭バイトの MSB が 1 でない場合はデータが壊れているので読み捨て
221 while (uart.available())
222 {
223 uart.read();
224 }
225 return false;
226 }
227
228 // フッター読み捨て
229 for (int i = 0; i < FooterSize; ++i)
230 {
231 uart.read();
232 }
233
234 // バッファに残っているデータを読み捨て
235 while (uart.available())
236 {
237 uart.read();
238 }
239
240 rxNode->transmitMs = millis();
241
242 return true;
243 }

245 void twoWayUpdate()
246 {
247 // transTime =[内部処理時間:10~20ms]+[キャリアセンス:初回5.2ms 連続通信時0.5ms]+[不要データの通信:3.2ms]+[バイトごとの送信時間:0.16ms]
248 const double sendTime = 52.0;
249 const double receiveTime = 52.0;
250
251 static uint32_t lastTransmitMs = 0;
252
253 if (millis() - lastTransmitMs > sendTime + receiveTime && receiveUpdate())
254 {
255 sendUpdate();
256 lastTransmitMs = millis();
257 }
258 else
259 {
260 if (millis() - lastTransmitMs > (uint32_t)random(2000))
261 {
262 sendUpdate();
263 lastTransmitMs = millis();
264 }
265 }

268 bool isTimeout(uint32_t timeoutMs) const
269 {
270 switch (getTransmitMode())
271 {
272 case TransmitMode::Send:
273 return millis() - txNode->transmitMs > timeoutMs;
274 case TransmitMode::Receive:
275 return millis() - rxNode->transmitMs > timeoutMs;
276 case TransmitMode::TwoWay:
277 return millis() - txNode->transmitMs > timeoutMs || millis() - rxNode->transmitMs > timeoutMs;
278 case TransmitMode::Empty:
279 return false;
280 default:
281 return false;
282 }
283 }

285 void RestartOnTimeout()
286 {
287 if (not isTimeout(2000))
288 {
289 return;
290 }
291
292 // 通信が不通になった場合はソフトウエアリセット
293 if (millis() - lastRestartMs > 2000)
294 {
295 waitUntilCommandAccept();
296 uart.print("SRST\r\n");
297 lastRestartMs = millis();
298 while (uart.available())
299 {
300 uart.read();
301 }
302 }
303 }
304
306 void waitUntilCommandAccept()
307 {
308 if (busyPin)
309 {
310 lastWaitUntilCommandAcceptMs = millis();
311
312 while (digitalRead(*busyPin)) // busyピンがHIGHの間はコマンドを受け付けない
313 {
314 delayMicroseconds(10); // チャタリング防止
315
316 if (millis() - lastWaitUntilCommandAcceptMs > 200)
317 {
318 break; // タイムアウトした場合は強制的にコマンドを受け付ける
319 }
320 }
321 }
322 else
323 {
324 delay(200); // busy pinが設定されていない場合は200ms待つ
325 }
326
327 // 待機している間に余計なデータが送られてきている可能性があるので読み捨てる
328 while (uart.available())
329 {
330 uart.read();
331 }
332 }
