150 template <size_t N>
151 std::array<double, N> toOmni() const
152 {
153 uint8_t powerLimit = 255;
154 uint8_t turnPowerLimit = 255;
155 const auto mappedTurn = Udon::Map(turn, -255, 255, -turnPowerLimit, turnPowerLimit);
156
157 std::array<double, 4> powers{ {
158 +vector.x + -vector.y + mappedTurn,
159 -vector.x + -vector.y + mappedTurn,
160 -vector.x + +vector.y + mappedTurn,
161 +vector.x + +vector.y + mappedTurn,
162 } };
163
164 // 上で算出した出力値は {powerLimit} を超える場合があります。
165 // 超えた場合、最大出力のモジュールの出力を {powerLimit} として他のモジュールの出力を圧縮します。
166 auto&& max = abs(*std::max_element(powers.begin(), powers.end(), [](double lhs, double rhs)
167 { return abs(lhs) < abs(rhs); }));
168
169 if (max > powerLimit)
170 {
171 const auto ratio = powerLimit / max;
172
173 std::transform(powers.begin(), powers.end(), powers.begin(),
174 [ratio](double power)
175 { return power * ratio; });
176 }
177
178 // todo
179 return powers;
180 }

193 template <size_t WheelCount = 4>
194 std::array<Udon::Polar, WheelCount> toSteer(uint8_t powerLimit = 255,
195 uint8_t turnPowerLimit = 255) const
196 {
197 static_assert(WheelCount >= 3, "WheelCount must be greater than or equal to 3.");
198 static_assert(WheelCount == 4, "Not supported over 4 wheels now."); // todo: 5輪以上対応
199
200 // 旋回移動ベクトル
201 const Udon::Vec2 turnVec = { 0, Udon::Map(this->turn, -255, 255, -turnPowerLimit, turnPowerLimit) };
202
203 // 旋回移動ベクトル、進行移動ベクトルから角車輪の進行方向ベクトルを算出
204 std::array<Udon::Vec2, WheelCount> vectors;
205
206 for (size_t i = 0; i < WheelCount; ++i)
207 {
208 vectors.at(i) = vector + turnVec.rotated(Udon::Pi * (i * 2 + 3) / WheelCount);
209 }
210
211 // 方向ベクトルから極座標系(theta:旋回角, r:タイヤ出力)に変換
212 std::array<Udon::Polar, WheelCount> modules;
213
214 for (size_t i = 0; i < WheelCount; ++i)
215 {
216 modules.at(i) = vectors.at(i).toPolar();
217 }
218
219 // 上で算出した出力値は {limit} を超える場合があります。
220 // 超えた場合、最大出力のモジュールの出力を {limit} として他のモジュールの出力を圧縮します。
221 // 出力値はベクトルの長さから求めるため負にならない {max} は正の値であることが保証
222 auto&& max = *std::max_element(modules.begin(), modules.end(), [](const Udon::Polar& lhs, const Udon::Polar& rhs)
223 { return lhs.r < rhs.r; });
224
225 if (max.r > powerLimit)
226 {
227 const auto ratio = powerLimit / max.r;
228 for (auto&& module : modules)
229 {
230 module.r *= ratio;
231 }
232 }
233
234 return modules;
235 }
