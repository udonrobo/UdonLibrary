192  template <size_t WheelCount = 4>
193  std::array<Udon::Polar, WheelCount> toSteer(uint8_t powerLimit = 255,
194  uint8_t turnPowerLimit = 255) const
195  {
196  static_assert(WheelCount >= 3, "WheelCount must be greater than or equal to 3.");
197  static_assert(WheelCount == 4, "Not supported over 4 wheels now."); // todo: 5輪以上対応
198 
199  // 旋回移動ベクトル
200  const Udon::Vec2 turnVec = { 0, Udon::Map(this->turn, -255, 255, -turnPowerLimit, turnPowerLimit) };
201 
202  // 旋回移動ベクトル、進行移動ベクトルから角車輪の進行方向ベクトルを算出
203  std::array<Udon::Vec2, WheelCount> vectors;
204 
205  for (size_t i = 0; i < WheelCount; ++i)
206  {
207  vectors.at(i) = vector + turnVec.rotated(Udon::Pi * (i * 2 + 3) / WheelCount);
208  }
209 
210  // 方向ベクトルから極座標系(theta:旋回角, r:タイヤ出力)に変換
211  std::array<Udon::Polar, WheelCount> modules;
212 
213  for (size_t i = 0; i < WheelCount; ++i)
214  {
215  modules.at(i) = vectors.at(i).toPolar();
216  }
217 
218  // 上で算出した出力値は {limit} を超える場合があります。
219  // 超えた場合、最大出力のモジュールの出力を {limit} として他のモジュールの出力を圧縮します。
220  // 出力値はベクトルの長さから求めるため負にならない {max} は正の値であることが保証
221  auto&& max = *std::max_element(modules.begin(), modules.end(), [](const Udon::Polar& lhs, const Udon::Polar& rhs)
222  { return lhs.r < rhs.r; });
223 
224  if (max.r > powerLimit)
225  {
226  const auto ratio = powerLimit / max.r;
227  for (auto&& module : modules)
228  {
229  module.r *= ratio;
230  }
231  }
232 
233  return modules;
234  }
