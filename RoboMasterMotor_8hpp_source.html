87 void setCurrent(int16_t current)
88 {
89 current = current * directionSign();
90
91 const auto transmitCurrent = static_cast<int16_t>(current * 16384 / 20000); // 16bit符号なし整数に変換
92 const auto motorIndex = (motorId - 1) * 2;
93
94 nodeTx->data[motorIndex + 0] = transmitCurrent >> 8 & 0xff; // high byte
95 nodeTx->data[motorIndex + 1] = transmitCurrent >> 0 & 0xff; // low byte
96 }

128 static void onReceive(void* p)
129 {
130 auto self = static_cast<RoboMasterBase*>(p);
131
132 // 変化角を計算
133 const auto currentAngle = self->nodeRx->data[0] << 8 | self->nodeRx->data[1];
134 const auto dTheta = self->angle - currentAngle;
135 self->angle = currentAngle;
136
137 // 変化量がいきなり半周を超えた -> 計算値が-π~π間を通過 -> 一周分オフセットを加減算
138 if (dTheta > ppr / 2)
139 {
140 self->offsetAngle += ppr;
141 }
142 else if (dTheta < -ppr / 2)
143 {
144 self->offsetAngle -= ppr;
145 }
146 }
147 };
