64 void update(double controlValue, double targetValue) noexcept
65 {
66 // 係数の適用
67 const auto coefficient = requestConstant ? *requestConstant : constant;
68 requestConstant.reset();
69
70 // 偏差の計算
71 const double error = targetValue - controlValue;
72
73 // 比例量の計算
74 power.p = (error - prevError) * coefficient.p;
75
76 // 積分量の計算
77 power.i = error * coefficient.i * INTERVAL_S;
78
79 // 微分量の計算
80 power.d = (error - 2 * prevError + twoPrevError) / INTERVAL_S;
81
82 // ローパスフィルタの計算
83 rowPassDif += (power.d - rowPassDif) / 8 * coefficient.d;
84
85 const double deltaPower =Udon::Constrain( power.p + power.i + rowPassDif,-MAX_POWER/2.0, MAX_POWER/2.0);
86
87 output = Udon::Constrain(output + deltaPower, -MAX_POWER, MAX_POWER);
88
89 // 偏差の保存
90 twoPrevError = prevError;
91 prevError = error;
