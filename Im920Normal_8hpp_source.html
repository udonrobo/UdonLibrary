152  void sendUpdate()
153  {
154  uart.print("TXDA ");
155 
156  Udon::BitPack(txNode->data, txNode->data + txNode->size, [this](uint8_t data)
157  { uart.write(data); });
158 
159  uart.print("\r\n");
160  }

162  bool receiveUpdate()
163  {
164  // header = [Dummy: 2byte] + [,] + [Node number: 4byte] + [,] + [RSSI: 2byte] + [:]
165  constexpr int HeaderSize = 2 + 1 + 4 + 1 + 2 + 1;
166 
167  // footer = [\r] + [\n]
168  constexpr int FooterSize = 1 + 1;
169 
170  // FrameSize = header + data + footer
171  const int FrameSize = HeaderSize + Udon::BitPackedSize(rxNode->size) + FooterSize;
172 
173  // 受信バッファにデータがない場合は何もしない
174  if (uart.available() < FrameSize)
175  {
176  return false;
177  }
178 
179  // ヘッダー読み捨て
180  for (int i = 0; i < HeaderSize; ++i)
181  {
182  uart.read();
183  }
184 
185  // データ読み込み
186  if (not Udon::BitUnpack(rxNode->data, rxNode->data + rxNode->size, [this]() -> uint8_t
187  { return uart.read(); }))
188  {
189  // 先頭バイトの MSB が 1 でない場合はデータが壊れているので読み捨て
190  while (uart.available())
191  {
192  uart.read();
193  }
194  return false;
195  }
196 
197  // フッター読み捨て
198  for (int i = 0; i < FooterSize; ++i)
199  {
200  uart.read();
201  }
202 
203  // バッファに残っているデータを読み捨て
204  while (uart.available())
205  {
206  uart.read();
207  }
208 
209  rxNode->transmitMs = millis();
210 
211  return true;
212  }

214  void twoWayUpdate()
215  {
216  // transTime =[内部処理時間:10~20ms]+[キャリアセンス:初回5.2ms 連続通信時0.5ms]+[不要データの通信:3.2ms]+[バイトごとの送信時間:0.16ms]
217  const double sendTime = 10.0 + 5.2 + 3.2 + txNode->size * 0.16;
218  const double receiveTime = 10.0 + 5.2 + 3.2 + rxNode->size * 0.16;
219 
220  if (twoWayNum)
221  {
222  if (2 * millis() - (sendMitMs + receiveDeadTime) > sendTime + receiveTime)
223  { // 時間経過により再送信
224  sendUpdate();
225  }
226  if (receiveUpdate())
227  {
228  sendUpdate();
229  }
230  }
231  else
232  {
233  if (receiveUpdate())
234  {
235  sendUpdate();
236  }
237  }
238  }
